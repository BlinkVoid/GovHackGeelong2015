Google Maps APIs

marker within function initialize(), it can be animated
::
var marker=new google.maps.Marker({
  position:myCenter,
  animation:google.maps.Animation.BOUNCE
  icons: 'pinkball.png' \\it is possible to use other picture as the pin
  });

marker.setMap(map);
}

using pylilines
::
var myTrip = [stavanger,amsterdam,london]; //these locations are just coordinates
var flightPath = new google.maps.Polyline({
  path:myTrip,
  strokeColor:"#0000FF",
  strokeOpacity:0.8,
  strokeWeight:2
});

using polygon:
::
var flightPath = new google.maps.Polygon({
  path:myTrip,
  strokeColor:"#0000FF",
  strokeOpacity:0.8,
  strokeWeight:2,
  fillColor:"#0000FF",
  fillOpacity:0.4
});
